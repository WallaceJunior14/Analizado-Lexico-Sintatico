<calclist> ::= ε
             | <calclist> <stmt> EOL
             | <calclist> 'LET' 'NAME' '(' <symlist> ')' '=' <list> EOL
             | <calclist> error EOL

<stmt> ::= 'IF' <exp> 'THEN' <list>
         | 'IF' <exp> 'THEN' <list> 'ELSE' <list>
         | 'WHILE' <exp> 'DO' <list>
         | 'FOR' '(' <exp> ';' <exp> ';' <exp> ')' <list>
         | <exp>

<list> ::= ε
         | <stmt> ';' <list>

<exp> ::= <exp> CMP <exp>
        | <exp> '+' <exp>
        | <exp> '-' <exp>
        | <exp> '*' <exp>
        | <exp> '/' <exp>
        | <exp> AND <exp>
        | <exp> OR <exp>
        | '(' <exp> ')'
        | NUMBER
        | NAME
        | NAME '=' <exp>
        | FUNC '(' <explist> ')'
        | NAME '(' <explist> ')'

<explist> ::= <exp>
            | <exp> ',' <explist>

<symlist> ::= NAME
            | NAME ',' <symlist>
